Sentiment Analysis

lexicon-based

pos.words <- scan("positive-words.txt", what = "character")
neg.words <- scan("negative-words.txt", what = "character")

pos.words <- unlist(lapply(pos.words, function(pos) str_replace_all(pos, "[^a-zA-Z]+", "a")))
neg.words <- unlist(lapply(neg.words, function(neg) str_replace_all(neg, "[^a-zA-Z]+", "a")))

score.sentiment <- function(sentences, pos.words, neg.words, .progress='none')
{
  scores = laply(sentences, function(sentence, pos.words, neg.words) {
    
    sentence = gsub('[[:punct:]]', '', sentence) #!"#$%&'()*+,-./:;<=>?@[\]^_`{|}~.
    sentence = gsub('[[:cntrl:]]', '', sentence) #顺序值小于 32 或字符值为 127 的 ASCII 字符（控制字符）。控制字符包括换行符、换页符、退格符，等等。
    sentence = gsub("<.*?>", '', sentence)
    sentence = gsub('\\d+', '', sentence)
    sentence = gsub('\\s+', ' ', sentence)
    sentence = tolower(sentence)
    
    word.list = str_split(sentence, '\\s+')
    words = unlist(word.list)
    
    pos.matches = match(words, pos.words)
    neg.matches = match(words, neg.words)
    pos.matches = !is.na(pos.matches)
    neg.matches = !is.na(neg.matches)
    score = sum(pos.matches) - sum(neg.matches)
    pos = sum(pos.matches)
    neg = sum(neg.matches)
    return(c(score, pos, neg))
  }, pos.words, neg.words, .progress=.progress)#
  
  score = scores[,1]
  pos = scores[,2]
  neg = scores[,3]
  
  scores.df = data.frame(score=score, text=sentences)
  pos.df = data.frame(pos=pos, text=sentences)
  neg.df = data.frame(neg=neg, text=sentences)
  return(list(scores.df, pos.df, neg.df))
}

scores <- score.sentiment(review$ReviewText, pos.words, neg.words)
review$score <- scores[[1]]$score
review$pos <- scores[[2]]$pos
review$neg <- scores[[3]]$neg
